async function askQuestion(interaction) {
  const question = quizQuestions[currentQuestion];
  const questionEmbed = {
    embeds: [
      {
        title: `❓ سوال ${currentQuestion + 1}`,
        description: `${question.question}\n\n${question.options.join('\n')}`,
        color: 5763719,
        footer: { text: "30 ثانیه وقت داری! ⏳" }
      }
    ],
    components: [
      {
        type: 1,
        components: [
          { type: 2, style: 1, label: "A", custom_id: "answer_A" },
          { type: 2, style: 1, label: "B", custom_id: "answer_B" },
          { type: 2, style: 1, label: "C", custom_id: "answer_C" },
          { type: 2, style: 1, label: "D", custom_id: "answer_D" }
        ]
      }
    ]
  };
  await interaction.channel.send(questionEmbed);

  // نمایش امتیازات فعلی
  let scoresText = '';
  for (const playerId of quizPlayers) {
    const member = await interaction.guild.members.fetch(playerId);
    scoresText += `${member.user.username}: ${quizScores[playerId]} امتیاز\n`;
  }
  await interaction.channel.send({
    embeds: [
      {
        title: "📊 امتیازات فعلی",
        description: scoresText || 'هنوز امتیازی ثبت نشده!',
        color: 5763719
      }
    ]
  });
}

async function endQuiz(interaction) {
  let scoresText = '';
  let highestScore = 0;
  let winner = null;

  for (const playerId of quizPlayers) {
    const member = await interaction.guild.members.fetch(playerId);
    scoresText += `${member.user.username}: ${quizScores[playerId]} امتیاز\n`;
    if (quizScores[playerId] > highestScore) {
      highestScore = quizScores[playerId];
      winner = member.user.username;
    }
  }

  await interaction.channel.send({
    embeds: [
      {
        title: "🏆 پایان بازی!",
        description: `مسابقه تموم شد! برنده: **${winner}** با ${highestScore} امتیاز 🎉\n\n**جدول امتیازات:**\n${scoresText}`,
        color: 5763719
      }
    ]
  });

  // ریست بازی
  quizPlayers = [];
  quizScores = {};
  currentQuestion = 0;
  const resetEmbed = {
    embeds: [
      {
        title: "🧠 مسابقه اطلاعات عمومی",
        description: "دانشت رو به چالش بکش! تو این مسابقه باید به سوالات جواب بدی و امتیاز جمع کنی! 📚",
        color: 5763719,
        fields: [
          { name: "👥 تعداد بازیکنان", value: `${quizPlayers.length}/10`, inline: true },
          { name: "⏱️ زمان هر سوال", value: "30 ثانیه", inline: true },
          { name: "👤 حداقل بازیکنان", value: "3 نفر", inline: true },
          { name: "💰 جایزه بازی", value: "برنده 200 کوین 🤑", inline: true }
        ],
        footer: { text: "برای شرکت تو مسابقه، روی دکمه ورود کلیک کن! 🎮" }
      }
    ],
    components: [
      {
        type: 1,
        components: [
          { type: 2, style: 3, label: "ورود به بازی", custom_id: "join_quiz" },
          { type: 2, style: 1, label: "شروع بازی", custom_id: "start_quiz" },
          { type: 2, style: 2, label: "قوانین بازی 📜", custom_id: "quiz_rules" },
          { type: 2, style: 2, label: "بازگشت 🔙", custom_id: "back_to_menu" }
        ]
      }
    ]

  await quizMessage.edit(resetEmbed);

مدیریت بازی: بازی با دکمه‌ها کنترل می‌شه. کاربران می‌تونن وارد بازی بشن، سوالات رو جواب بدن و امتیازاتشون رو ببینن.

async function startRound(interaction) {
  const painterId = otoPlayers[currentPainter];
  const painter = await interaction.guild.members.fetch(painterId);
  const topic = otoTopics[currentRound];

  await painter.send(`موضوع این دور: **${topic}** 🎨\nلطفاً نقاشی خودت رو تو 2 دقیقه بکش و تو چت آپلود کن! ⏳`);

  await interaction.channel.send({
    embeds: [
      {
        title: `🎨 دور ${currentRound + 1}`,
        description: `${painter.user.username} نقاش این دوره! لطفاً منتظر آپلود نقاشی باشید... ⏳`,
        color: 15548997
      }
    ]
  });
}

async function startDrawRound(interaction) {
  const painterId = drawPlayers[currentPainter];
  const painter = await interaction.guild.members.fetch(painterId);
  const word = drawWords[currentRound];

  await painter.send(`کلمه این دور: **${word}** ✍️\nلطفاً نقاشی خودت رو تو 2 دقیقه بکش و تو چت آپلود کن! ⏳`);

  await interaction.channel.send({

}
