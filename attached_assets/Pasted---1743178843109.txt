```javascript
// Job.js
class Job {
  constructor(userId, jobType, income, cooldown) {
```

#### 2. ذخیره‌سازی (Storage.js)
```javascript
// Storage.js
const Job = require('./Job');

class Storage {
  constructor() {
    this.jobs = new Map();
    this.ccoin = new Map(); // برای ذخیره Ccoin کاربرا
  }

  getUserJob(userId) {
    return this.jobs.get(userId) || null;
  }

  assignJob(userId, jobType) {
    const jobData = {
      miner: { income: 200, cooldown: 12 * 60 * 60 * 1000 },
      trader: { income: 500, cooldown: 12 * 60 * 60 * 1000 },
      supporter: { income: 300, cooldown: 24 * 60 * 60 * 1000 },
      hunter: { income: 250, cooldown: 24 * 60 * 60 * 1000 },
      reporter: { income: 350, cooldown: 24 * 60 * 60 * 1000 },
      organizer: { income: 450, cooldown: 24 * 60 * 60 * 1000 },
      designer: { income: 320, cooldown: 24 * 60 * 60 * 1000 },
      guardian: { income: 280, cooldown: 24 * 60 * 60 * 1000 },
      streamer: { income: 380, cooldown: 24 * 60 * 60 * 1000 },
      guide: { income: 270, cooldown: 24 * 60 * 60 * 1000 },
      keeper: { income: 310, cooldown: 24 * 60 * 60 * 1000 },
      auditor: { income: 290, cooldown: 24 * 60 * 60 * 1000 },
      coach: { income: 340, cooldown: 24 * 60 * 60 * 1000 },
      coordinator: { income: 360, cooldown: 24 * 60 * 60 * 1000 },
    };
    const job = new Job(userId, jobType, jobData[jobType].income, jobData[jobType].cooldown);
    this.jobs.set(userId, job);
  }

  collectJobIncome(userId, taskProof) {
    const job = this.jobs.get(userId);
    if (job) {
      job.lastWorked = Date.now();
      job.xp += 10;
      job.lastTask = taskProof;
      const income = job.income;
      this.addCcoin(userId, income * 0.95); // 5% مالیات
      this.jobs.set(userId, job);
      return income;
    }
    return 0;
  }

  updateJobLevel(userId) {
    const job = this.jobs.get(userId);
    if (job && job.xp >= 50) {
      job.level += 1;
      job.xp = 0;
      job.income = Math.floor(job.income * 1.1);
      this.jobs.set(userId, job);
      return job.level;
    }
    return 0;
  }

  getUserCcoin(userId) {
    return this.ccoin.get(userId) || 0;
  }

  addCcoin(userId, amount) {
    const current = this.getUserCcoin(userId);
    this.ccoin.set(userId, current + amount);
  }

  spendCcoin(userId, amount) {
    const current = this.getUserCcoin(userId);
    if (current >= amount) {
      this.ccoin.set(userId, current - amount);
      return true;
    }
    return false;
  }
}

module.exports = new Storage();
```

```javascript
    if (now < job.lastWorked + job.cooldown) {
      const remaining = Math.ceil((job.lastWorked + job.cooldown - now) / 3600000);
      await interaction.update({ content: `⏰ هنوز ${remaining} ساعت مونده!` });
    } else {
      const tasks = {
        miner: '⛏️ یه پیام بامزه تو چت بفرست و لینک کن.',
        trader: '🏪 یه آیتم بخر یا بفروش و کد تراکنش بده.',
        supporter: '🤖 به یه سوال تو اطلاعات عمومی جواب بده (3+ 👍).',
        hunter: '🗺️ پیام رویداد امروز رو پیدا کن.',
        reporter: '📰 یه گزارش از سرور بنویس (2+ 👍).',
        organizer: '🎉 یه رویداد با 5 نفر برگزار کن.',
        designer: '🎲 یه چالش بساز (3 نفر شرکت کنن).',
        guardian: '🛡️ یه پیام مشکل‌دار گزارش بده.',
        streamer: '🎥 30 دقیقه استریم کن (3 شنونده).',
        guide: '👋 به یه تازه‌وارد کمک کن (1+ 👍).',
        keeper: '📊 حضور 5 نفر رو ثبت کن (✅).',
        auditor: '🔍 یه پیام مفید پیدا کن (3+ 👍).',
        coach: '🎮 یه راهنمای مینی‌گیم بنویس (2+ 👍).',
        coordinator: '💬 یه بحث با 10 پاسخ راه بنداز.',
      };
      await interaction.update({
        content: tasks[job.jobType] + '\nبعدش "ثبت کار" رو بزن.',
        components: [new MessageActionRow().addComponents(
          new MessageButton().setCustomId('jobs_submit').setLabel('ثبت کار').setStyle('SECONDARY')
        )],
      });
    }
  }

  if (customId === 'jobs_submit') {
    await interaction.reply({ content: '📩 لینک یا مدرک فعالیتت رو بفرست:', ephemeral: true });
    // Replit باید اینجا collector اضافه کنه
  }

  if (customId === 'jobs_upgrade') {
    const job = Storage.getUserJob(userId);
    if (!job || job.xp < 50) return await interaction.update({ content: '🎯 XP کافی نداری!' });
    const cost = job.level * 500 + 500;
    if (Storage.spendCcoin(userId, cost)) {
      const newLevel = Storage.updateJobLevel(userId);
      await interaction.update({ content: `🌟 شغلت به سطح ${newLevel} ارتقا یافت!` });
    } else {
      await interaction.update({ content: '💰 سکه کافی نداری!' });
    }
  }

  if (customId === 'jobs_status') {
    const job = Storage.getUserJob(userId);
    const desc = job ? `شغل: ${job.jobType} | سطح: ${job.level} | درآمد: ${job.income} | XP: ${job.xp}/50` : 'شغلی نداری!';
    await interaction.update({ content: desc });
  }

  if (customId.startsWith('select_')) {
    const jobType = customId.split('_')[1];
    const reqs = {
      miner: 0, trader: 10000, supporter: 5000, hunter: 5000, reporter: 7000,
      organizer: 12000, designer: 6000, guardian: 4000, streamer: 8000,
      guide: 3000, keeper: 5500, auditor: 4500, coach: 6500, coordinator: 7500,
    };
    if (Storage.getUserCcoin(userId) >= reqs[jobType]) {
      Storage.assignJob(userId, jobType);
      await interaction.update({ content: `✅ شغل ${jobType} انتخاب شد!`, components: [] });
    } else {
      await interaction.update({ content: '💰 پیش‌نیاز کافی نداری!' });
    }
  }
}

module.exports = { createJobsMenu, handleJobsInteraction };
```

```javascript
// index.js
const { Client, Intents } = require('discord.js');
const { createJobsMenu, handleJobsInteraction } = require('./JobsMenu');
const Storage = require('./Storage');

const client = new Client({ intents: [Intents.FLAGS.GUILDS, Intents.FLAGS.GUILD_MESSAGES] });

  console.log('ربات آماده‌ست!');
});

client.on('interactionCreate', async interaction => {
  if (!interaction.isButton()) return;

  if (interaction.customId === 'economy_jobs') {
    const userJob = Storage.getUserJob(interaction.user.id);
    await interaction.update(createJobsMenu(userJob));
  } else if (interaction.customId.startsWith('jobs_') || interaction.customId.startsWith('select_')) {
    await handleJobsInteraction(interaction);
  }
});

client.login('YOUR_BOT_TOKEN'); // توکن رباتت رو اینجا بذار
```

   ```json
     "dependencies": {
   ```

   بعد تو ترمینال Replit بزن: `npm install`

2. **توکن ربات:** تو `index.js` باید توکن رباتت رو بذاری.
